117  void normalize(){
118  normalized = true;
119  // looping on the filtered tokens and calculating the document frequency for each token
120  for(map<string, vector<int> >::const_iterator it = tftd2.begin();it != tftd2.end();it++){
121  string t = it->first;
122  int df = 0;
123  for(int j=0;j<tftd2[t].size();j++){
124  if(tftd2[t][j] > 0)
125  df++;
126  }
127  dft[t] = df;
128  }
129 
130  wtd.clear();
131  // looping on the filtered tokens and calculating the document weight wt,d for each token
132  for(map<string, vector<int> >::const_iterator it = tftd2.begin();it != tftd2.end();it++){
133  string t = it->first;
134  for(int j=0;j<tftd2[t].size();j++){
135  // formula from assignment
136  double s = indexersize();
137  double d = dft[t];
138  double tf = tftd2[t][j];
139  double temp1 = 1+((double)log(tf)/(double)log(10));
140  double temp2 = (double)log(s/d)/(double)log(10);
141  double Wtd = temp1*temp2;
142  if(Wtd == -INFINITY || Wtd == INFINITY || isnan(Wtd))
143  wtd[t].push_back(0);
144  else
145  wtd[t].push_back(Wtd);

167  friend const document & operator >>(document & d,indexer<size> & idx){
168  idx.normalized = false; // reading new document so indexer not normalized
169  idx.documents.push_back(d); // pushing document to documnet vector

171  int total_1 = 0;

173  tokenizer t;
174  vector<string> tokens = t.tokenize(cont); // changing the content of document into tokens
175 
176  // same as assignment 1 pushing zeros then incrementing the token
177  for(int i=0;i<tokens.size();i++){
178  string s = tokens[i];
179  if(idx.stpw(s)){
180  if(idx.tftd1[s].size() == 0){
181  for(int i=0;i<size;i++){
182  idx.tftd1[s].push_back(0);
183  }
184  }
185  idx.tftd1[s][idx.N]++;
186  total_1++;
187  }
188  else{
189  if(idx.tftd1[s].size() == 0){
190  for(int i=0;i<size;i++){
191  idx.tftd1[s].push_back(0);
192  }
193  }
194  idx.tftd1[s][idx.N]++;
195 
196  if(idx.tftd2[s].size() == 0){
197  for(int i=0;i<size;i++){
198  idx.tftd2[s].push_back(0);
